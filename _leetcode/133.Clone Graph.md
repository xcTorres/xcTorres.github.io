## Solution  
- Depth First Search  

```python
"""
# Definition for a Node.
class Node(object):
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

class Solution(object):
    def cloneGraph(self, node):
        """
        :type node: Node
        :rtype: Node
        """
        def dfs(node, clone):
            for neig in node.neighbors:
                if neig.val not in clone:
                    clone[neig.val] = Node(neig.val)
                    dfs(neig, clone)
                clone[node.val].neighbors.append(clone[neig.val])

        if not node:
            return None

        clone = {} 
        clone[node.val] = Node(node.val)
        dfs(node, clone)
        
        return clone[node.val]
```

- Breath First Search

```python
"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

class Solution:
    def cloneGraph(self, node: 'Node') -> 'Node':
        
        if not node:
            return None
        
        if not node.neighbors:
            return Node(node.val)
        
        clone = {}
        clone[node.val] = Node(node.val, None)
        q = []
        q.append(node)
        while q:
            cur = q.pop(0)
            
            for nei in cur.neighbors:
                if nei.val not in clone:
                    clone[nei.val] = Node(nei.val, None)
                    q.append(nei)
                clone[cur.val].neighbors.append(clone[nei.val])
            
        return clone[node.val]
```
