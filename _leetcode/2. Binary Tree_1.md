## Preorder Traversal:
- Recursive
``` python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def preorderTraversal(self, root: TreeNode) -> List[int]:
        result = []
        self.dfs(result, root)
        return result
    
    def dfs(self, result, root):
        if not root:
            return
        result.append(root.val)
        self.dfs(result, root.left)
        self.dfs(result, root.right)  
```
        
- Iterative:   
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def preorderTraversal(self, root: TreeNode) -> List[int]:
        
        if not root:
            return []
        
        result = []
        stk = []
        node = root
        while stk or node:
            while node:
                result.append(node.val)
                stk.append(node)
                node = node.left
            node = stk[-1]
            stk.pop()
            node = node.right
        
        return result  
```

## Postorder Traversal:  

- Recursive:

```python
# Definition for a binary tree node.

#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def postorderTraversal(self, root: TreeNode) -> List[int]:
        result = []
        self.dfs(result, root)
        return result
    
    def dfs(self, result, root):
        if not root:
            return

```

- Iterative:  
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def postorderTraversal(self, root: TreeNode) -> List[int]:
        
        if not root:
            return []
        
        result = []
        stk = []
        node = root
        last_visited = root
        while stk or node:
            while node:
                stk.append(node)
                node = node.left
            node = stk[-1]
            if not node.right or last_visited == node.right:
                result.append(node.val)
                last_visited = node
                stk.pop()
                node = None
            else:
                node = node.right
        
        return result  
```
